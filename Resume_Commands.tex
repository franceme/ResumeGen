\newcommand*{\draft}{%
  \newlength\testh
  \newlength\testw
  \newcommand{\shorttextwatermark}{draft  \\\     }
  \setlength{\testh}{\heightof{\shorttextwatermark}}
  \setlength{\testw}{\widthof{\shorttextwatermark}}
  \newwallpaper[allpages,tilexsize=8\testw,tileysize=8\testh,boxalign=left]{\shorttextwatermark}
}
